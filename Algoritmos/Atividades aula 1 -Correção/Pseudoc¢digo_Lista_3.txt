INICIO DO ALGORITMO
   QP, PP : INTEIRO;
   VALOR, PAGAS, SALDO : REAL;
   MOSTRE("INFORME A QUANTIDADE DE PRESTAÇÕES: ");
   LER (QP);
   MOSTRE("INFORME A QUANTIDADE DE PRESTAÇÕES PAGAS: ");
   LER (PP);
   MOSTRE("INFORME O VALOR DE CADA PRESTAÇÃO: ");
   LER (VALOR);
   PAGAS = VALOR * PP;
   SALDO = (QP * VALOR) - PAGAS;
   MOSTRE("VALOR JÁ PAGO: ", PAGAS);
   MOSTRE("SALDO DEVEDOR: ", SALDO);
FIM DO ALGORITMO
